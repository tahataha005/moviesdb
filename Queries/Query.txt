FOR MAX EARNINGS RANK QUERY:

SELECT P.NAME, M.EARNINGS_RANK
FROM PERSON P,MOVIES M,OSCAR O
WHERE P.ID = OSCAR.PERSON_ID AND M.ID = O.MOVIE_ID AND O.TYPE = "BEST_PICTURE"
AND EARNINGS_RANK = (SELECT MAX(EARNINGS_RANK)
FROM MOVIES)


FOR MIN EARNINGS RANK QUERY:

SELECT P.NAME, M.EARNINGS_RANK
FROM PERSON P,MOVIES M,OSCAR O
WHERE P.ID = OSCAR.PERSON_ID AND M.ID = O.MOVIE_ID AND O.TYPE = "BEST_PICTURE"
AND EARNINGS_RANK = (SELECT MIN(EARNINGS_RANK)
FROM MOVIES)
